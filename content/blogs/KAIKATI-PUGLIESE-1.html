2) Mélange d'agents aléatoires et stratégiques¶

Stratégie conditionnelle ($a_{1,t}=+sgn(r_t)$)¶

In [3]:
phi_1 = 1
r_0 = np.random.normal()

def sign(x):
    return int(x/abs(x))

p_t = [100]
p_t_sans_agent = [100]
r_t = [r_0]
r_t_variance, r_t_esperance = 0,r_0/NIT

time = list(range(NIT))
for t in range(1,NIT):
    w_1_t = phi_1 * sign(r_t[t-1])
    r_t.append(epsilons[t] + w_1_t)
    p_t.append(p_t[-1] + r_t[-1])
    p_t_sans_agent.append(p_t_sans_agent[-1] + epsilons[t])
    
    r_t_esperance += r_t[-1]/NIT
    r_t_variance += (r_t[-1]**2)/NIT

r_t_variance -= r_t_esperance**2

autocorrelation_sans_agents = []
autocorrelation = []
for l in range(31):
    covariance_l_agents = 0
    covariance_l = 0
    r_t_esperance_sans_agent = 0
       
    for j in range(len(r_t)):
        r_t_esperance_sans_agent += epsilons[j]/len(r_t)
    
    for i in range(len(r_t)-l):
        covariance_l_agents += (r_t[i]-r_t_esperance)*(r_t[i+l]-r_t_esperance)/(len(r_t)-l)
        covariance_l += (epsilons[i]-r_t_esperance_sans_agent)*(epsilons[i+l]-r_t_esperance_sans_agent)/(len(r_t)-l)
        
    autocorrelation.append(covariance_l_agents)
    autocorrelation_sans_agents.append(covariance_l)

figure = plt.figure(figsize=(20,3))
    
ax1 = figure.add_subplot(131)
ax1.set_title("Price Evolution (Conditional Strategy)")
ax1.plot(time,p_t,label="Log Price ($\phi_1$ = " + str(phi_1) + ")")
ax1.plot(time,p_t_sans_agent,label="Log Price ($\phi_1$ = 0)")

plt.legend()

ax2 = figure.add_subplot(132)
ax2.plot(time, r_t, label="Rendements")
ax2.set_title('Rendements (Conditional Strategy)')
ax2.plot(time,[r_t_esperance]*len(time),label="Expected value (1 agent)")

ax2.plot(time,[r_t_esperance + np.sqrt(r_t_variance)]*len(time),label="+ 1 $\sigma$",color='r')
ax2.plot(time,[r_t_esperance - np.sqrt(r_t_variance)]*len(time),label="- 1 $\sigma$",color='r')

plt.legend()

ax3 = figure.add_subplot(133)
ax3.set_title("Autocorrelogram (Conditional Strategy)")
ax3.scatter(range(31),autocorrelation, label="Autocorrelation (1 agent)")
ax3.scatter(range(31),autocorrelation_sans_agents, label="Autocorrelation")

plt.legend()
plt.grid()

Observations :

  • Les prix obtenus dans le cadre de la stratgie conditionnelle prsentent les mmes variations que les prix alatoires mais avec une plus grande magnitude (il faudrait cependant effectuer de nombreux tirages pour vrifier que ce comportement est significatif) ; il faut galement remarquer que les rendements prsentent des oscillations qui paraissent assez rgulires en fonction du temps (priode aux alentours de la centaine d'units de temps).
  • On constate que l'autocorrlation dcrot quasiment en $x^{-1}$ avec 1 agent alors qu'elle est presque nulle ds un pas de 1 dans le cas purement alatoire : on peut affirmer que les rendements sont devenus nettement plus prvisibles (autocorrlation ngligeable partir d'un pas de 15 seulement).

#initialisations
phi_1 = 1
r_0 = np.random.normal()
k = 7

def sign(x):
    return int(x/abs(x))

def suivi_tendance_k(p_t,k):
    # On vrifie qu'on a assez de points pour pouvoir faire la moyenne mobile
    test = len(p_t) - k
    
    if test >=0: # On a assez de points
        MA_k = 0
        for i in range(len(p_t)-k,len(p_t)):
            MA_k += p_t[i] / k

        return sign(p_t[-1] - MA_k)
    
    else: # On n'a pas assez de points, mais on fait avec : on calcule une moyenne uniquement avec les points disponibles
        moy = 0
        for i in range(0,len(p_t)):
            moy += p_t[i]/len(p_t)
        return moy

#stratgie conditionnelle
p_t = [100]
p_t_sans_agent = [100]
r_t = [r_0]
r_t_variance, r_t_esperance = 0,r_0/NIT

time = list(range(NIT))
for t in range(1,NIT):
    w_1_t = phi_1 * suivi_tendance_k(r_t,k)

        w_1_t = phi_1 * sign(r_t[i][t-1])
